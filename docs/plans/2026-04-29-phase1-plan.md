# RecipeKeepsake — Phase 1 Build Plan

**Goal:** Build the recipe blog (home + recipe page) and mobile recording UI with image generation, audio storage, and WhatsApp share.
**Layer:** Multi-layer — Python prompts + FastAPI + Next.js UI
**Architecture:** Python handles all AI work (transcription, translation, structuring, image generation) via FastAPI on port 8080. Next.js on port 3000 handles UI — reads recipes from Supabase directly, sends audio to FastAPI for processing. Clean separation: swap either side without touching the other.
**Design doc:** `docs/plans/2026-04-29-phase1-ui-design.md`

```python
# prompts/image.py
from openai import OpenAI

IMAGE_PROMPT_TEMPLATE = (
    "A beautiful, appetizing close-up photograph of {dish_name}, "
    "a traditional South Indian dish. "
    "Warm natural lighting, rustic wooden surface, authentic home-cooked presentation. "
    "Vibrant colours, sharp focus. No text, no watermarks, no people."
)


def generate_dish_image(dish_name: str) -> str:
    """Call DALL-E 3 with dish name. Returns the generated image URL.
    Note: URL expires after ~1 hour — caller must download and re-upload to Supabase Storage.
    """
    client = OpenAI()
    prompt = IMAGE_PROMPT_TEMPLATE.format(dish_name=dish_name)
    response = client.images.generate(
        model="dall-e-3",
        prompt=prompt,
        size="1024x1024",
        quality="standard",
        n=1,
    )
    return response.data[0].url
```

```python
# Add to scripts/serve.py

@app.post("/generate-image")
async def generate_image_endpoint(body: dict):
    """Generate a DALL-E image for a dish name. Returns permanent Supabase Storage URL."""
    dish_name = body.get("dish_name", "Indian dish")

    from prompts.image import generate_dish_image
    dalle_url = generate_dish_image(dish_name)

    # Download image and upload to Supabase Storage if configured
    if os.environ.get("SUPABASE_URL") and os.environ.get("SUPABASE_SERVICE_KEY"):
        import httpx
        from supabase import create_client
        img_bytes = httpx.get(dalle_url).content
        filename = f"{dish_name.lower().replace(' ', '-')}-{int(__import__('time').time())}.png"
        sb = create_client(os.environ["SUPABASE_URL"], os.environ["SUPABASE_SERVICE_KEY"])
        sb.storage.from_("images").upload(filename, img_bytes, {"content-type": "image/png"})
        public_url = sb.storage.from_("images").get_public_url(filename)
        return JSONResponse(content={"image_url": public_url})

    # Dev fallback: return DALL-E URL directly (expires in ~1hr)
    return JSONResponse(content={"image_url": dalle_url})


@app.get("/recipes")
async def list_recipes():
    """Return all recipes for the blog home page."""
    if not (os.environ.get("SUPABASE_URL") and os.environ.get("SUPABASE_SERVICE_KEY")):
        return JSONResponse(content={"recipes": []})
    from tools.storage import list_recipes as _list
    return JSONResponse(content={"recipes": _list()})
```

#### Chunk 5.1 — Record button + pipeline progress

Files:
- Create: `web/nextjs/app/record/page.tsx` (client component)

```tsx
'use client'
import { useState, useRef } from 'react'

type Stage = 'idle' | 'recording' | 'transcribing' | 'translating' | 'structuring' | 'imaging' | 'done' | 'error'

const STAGE_LABELS: Record<string, string> = {
  transcribing: 'Transcribing voice...',
  translating:  'Translating to English...',
  structuring:  'Building recipe...',
  imaging:      'Generating image...',
}

export default function RecordPage() {
  const [stage, setStage] = useState<Stage>('idle')
  const [seconds, setSeconds] = useState(0)
  const [photo, setPhoto] = useState<File | null>(null)
  const [recipe, setRecipe] = useState<any>(null)
  const [error, setError] = useState('')
  const mediaRecorderRef = useRef<MediaRecorder | null>(null)
  const chunksRef = useRef<Blob[]>([])
  const timerRef = useRef<ReturnType<typeof setInterval> | null>(null)

  const API = process.env.NEXT_PUBLIC_API_URL ?? 'http://localhost:8080'

  function fmt(s: number) {
    return `${String(Math.floor(s/60)).padStart(2,'0')}:${String(s%60).padStart(2,'0')}`
  }

  async function startRecording() {
    setError('')
    setRecipe(null)
    chunksRef.current = []
    setSeconds(0)

    const stream = await navigator.mediaDevices.getUserMedia({ audio: true })
    const mr = new MediaRecorder(stream)
    mediaRecorderRef.current = mr
    mr.ondataavailable = e => chunksRef.current.push(e.data)
    mr.onstop = handleStop
    mr.start()
    setStage('recording')

    timerRef.current = setInterval(() => setSeconds(s => {
      if (s >= 599) { mr.stop(); return s }
      return s + 1
    }), 1000)
  }

  function stopRecording() {
    clearInterval(timerRef.current!)
    mediaRecorderRef.current?.stop()
  }

  async function handleStop() {
    const blob = new Blob(chunksRef.current, { type: 'audio/webm' })
    const form = new FormData()
    form.append('audio', blob, 'recording.webm')
    if (photo) form.append('photo', photo)

    try {
      setStage('transcribing')
      const res = await fetch(`${API}/capture`, { method: 'POST', body: form })
      const data = await res.json()
      if (!res.ok) throw new Error(data.detail ?? 'Pipeline failed')
      setRecipe(data)
      setStage('done')
    } catch (e: any) {
      setError(e.message)
      setStage('error')
    }
  }

  // ── REVIEW SCREEN (after pipeline) ──
  if (stage === 'done' && recipe) {
    return <ReviewScreen recipe={recipe} apiUrl={API} />
  }

  // ── PROGRESS SCREEN ──
  if (['transcribing','translating','structuring','imaging'].includes(stage)) {
    return (
      <div className="flex flex-col items-center gap-4 py-16">
        <div className="w-10 h-10 border-4 border-[#1E1B4B] border-t-[#A78BFA] rounded-full animate-spin" />
        <p className="text-[#818CF8]">{STAGE_LABELS[stage]}</p>
        <div className="text-xs text-[#475569] space-y-1 text-center">
          {Object.entries(STAGE_LABELS).map(([s, label]) => (
            <p key={s} className={stage === s ? 'text-[#A78BFA]' : 'text-[#334155]'}>
              {stage === s ? '⏳' : '○'} {label}
            </p>
          ))}
        </div>
      </div>
    )
  }

  // ── IDLE / RECORD SCREEN ──
  return (
    <div className="flex flex-col items-center gap-6 py-8">
      {/* Optional photo */}
      <label className="flex items-center gap-2 text-sm text-[#64748B] cursor-pointer">
        <span className="bg-[#0F0F23] border border-[#1E1B4B] rounded-xl px-4 py-2">
          📷 {photo ? photo.name : 'Add a photo (optional)'}
        </span>
        <input type="file" accept="image/*" className="hidden"
          onChange={e => setPhoto(e.target.files?.[0] ?? null)} />
      </label>

      {/* Record button */}
      <button
        onClick={stage === 'recording' ? stopRecording : startRecording}
        className={`w-24 h-24 rounded-full text-3xl text-white border-none transition-all ${
          stage === 'recording'
            ? 'bg-red-500 animate-pulse shadow-[0_0_20px_rgba(239,68,68,0.5)]'
            : 'bg-[#A78BFA] hover:scale-105'
        }`}
      >
        {stage === 'recording' ? '⏹️' : '🎙️'}
      </button>

      {stage === 'recording'
        ? <p className="text-2xl font-mono text-[#94A3B8]">{fmt(seconds)}</p>
        : <p className="text-[#64748B] text-sm">Tap to start recording</p>
      }

      {error && (
        <p className="text-red-400 text-sm text-center">❌ {error}</p>
      )}
    </div>
  )
}
```

```tsx
      </div>

      {/* Ingredients (read-only summary) */}
      <div>
        <label className="text-xs text-[#818CF8] uppercase tracking-wider">
          Ingredients ({recipe.ingredients?.length ?? 0})
        </label>
        <div className="mt-1 bg-[#0F0F23] border border-[#1E1B4B] rounded-xl px-4 py-2 text-sm text-[#94A3B8]">
          {(recipe.ingredients ?? []).map((i: any) => i.item).join(', ')}
        </div>
      </div>

      {/* Cook notes */}
      <div>
        <label className="text-xs text-[#818CF8] uppercase tracking-wider">Grandma's notes</label>
        <textarea value={notes} onChange={e => setNotes(e.target.value)} rows={3}
          className="w-full mt-1 bg-[#0F0F23] border border-[#1E1B4B] rounded-xl px-4 py-2 text-[#E2E8F0] text-sm resize-none" />
      </div>

      {/* Review flags */}
      {(recipe.review_flags ?? []).length > 0 && (
        <div className="bg-red-950/30 border border-red-900/40 rounded-xl p-3">
          <p className="text-xs text-red-400 font-semibold mb-1">⚠️ Review flags</p>
          {recipe.review_flags.map((f: string, i: number) => (
            <p key={i} className="text-xs text-red-300">{f}</p>
          ))}
        </div>
      )}

      {/* Share button — URL pre-built, no await before window.open (iOS Safari) */}
      {waUrl && !saved && (
        <button
          onClick={() => { setSaved(true); window.open(waUrl) }}
          className="w-full bg-[#A78BFA] text-white py-3 rounded-2xl font-semibold mt-2">
          Save & Share on WhatsApp 💬
        </button>
      )}

      {saved && (
        <div className="text-center text-sm text-[#64748B]">
          ✅ Saved. <a href="/" className="text-[#818CF8]">Back to all recipes</a>
        </div>
      )}
    </div>
  )
}
```
